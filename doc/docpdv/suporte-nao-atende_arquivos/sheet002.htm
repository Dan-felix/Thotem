select 
  pool.idhotel,
  pool.nome as nomehotel,
  pool.razaosocial as razaosocialhotel,
  pool.cnpj as cnpjhotel,
  pool.idempcondominio as idcondo,
  cond.nome as nomecond,
  cond.razaosocial as razaosocialcond,
  cond.cnpj as cnpjcond
from
(select
  ht.idhotel,
  ht.idempcondominio,
 

 

 

 

  pe.nome,
  pe.razaosocial,
  pe.numdocumento as cnpj
from
  hotel ht
inner join
  pessoa pe on ht.idhotel = pe.idpessoa
inner join
  empresaprop ep on ht.idhotel = ep.idpessoa and pe.idpessoa = ep.idpessoa
inner join
  paramhotel ph on ht.idhotel=ph.idhotel
inner join
  paramcentres pc on ht.idhotel=pc.idhotel
where
    pc.flgativo='T' 
and ht.idhotel not in (-1,900,2048)) pool,
(select
    pe.idpessoa,
    pe.nome,
    pe.razaosocial,
    pe.numdocumento as cnpj
  from
    pessoa pe
  where idpessoa in (
    select
      ht.idempcondominio
    from
      hotel ht
    inner join
      pessoa pe on ht.idhotel = pe.idpessoa
    inner join
      empresaprop ep on ht.idhotel = ep.idpessoa and pe.idpessoa = ep.idpessoa
    inner join
      paramhotel ph on ht.idhotel=ph.idhotel
    inner join
      paramcentres pc on ht.idhotel=pc.idhotel
    where
        pc.flgativo='T' 
    and ht.idhotel not in (-1,900,2048))) Cond
where
  pool.idempcondominio=cond.idpessoa(+)
order by 
 

  pool.nome,
  cond.idpessoa
